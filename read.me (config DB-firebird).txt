Comandos para criar o banco de dados.

CREATE DATABASE 'C:\caminho\banco_peixaria.fdb'
USER 'admin' PASSWORD 'admin12345'
PAGE_SIZE 8192;

-- Criação da tabela de produtos
CREATE TABLE produtos (
    id_produto INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    quantidade_kg NUMERIC(10, 2) NOT NULL
);

-- Criação de auditoria para rastrear alterações
CREATE TABLE auditoria (
    id_auditoria INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    operacao VARCHAR(50) NOT NULL, -- "adicionar" ou "retirar"
    nome_produto VARCHAR(100),
    quantidade_kg NUMERIC(10, 2),
    data_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
